图论最短路的几种算法刨析及比较（C++）

单源最短路算法

朴素dijkstra算法

以下所有讨论的约定：n为点数，m为边数。所有的图都采用静态模拟的形式，即邻接表或者邻接矩阵。

那什么是dijkstra算法呢？

迪杰斯特拉算法(Dijkstra)是由荷兰计算机科学家狄克斯特拉于1959年提出的，因此又叫狄克斯特拉算法。是从一个顶点到其余各顶点的最短路径算法，解决的是有权图中最短路径问题。迪杰斯特拉算法主要特点是从起始点开始，采用贪心算法的策略，每次遍历到始点距离最近且未访问过的顶点的邻接节点，直到扩展到终点为止。（百度百科）

dijkstra算法的核心思想就是进行n次迭代去确定每个点到起点的最小值，最后输出的终点即为我们要找的最短路的距离。

需要额外注意一点的是，dijkstra算法并不适用于存在负权边的情况。

因为dijkstra算法实际上是采用贪心的策略，我们要找到确定的最短路径加入集合，如果存在负边权，就有可能更新出更短的距离。举个简单的例子：

比如下图，我们想从A点走到C点，dijkstra算法会选择直接从A走到C，而不会考虑A->B->C这条路。因为它只会考虑局部最优解，但在存在负值边的情况下，局部最优解并不是全局最优解！

代码如下，题目参考acwing849.

 #include<iostream>
 #include <cstring>
 #include<cstdio>
 using namespace std;
 const int N=510;
 int g[N][N],dist[N],n,m;
 bool st[N];
 int Dijkstra(){
     memset(dist,0x3f,sizeof dist);
     dist[1]=0;
     
     for(int i=0;i<n;i++){    //n次迭代    覆盖到所有结点 
         int t=n+1;
         for(int j=1;j<=n;j++){    //找出未确定节点中距离最短的一个 
             if(!st[j]&&dist[t]>dist[j])
             t=j;
         }
        st[t]=true;
        //利用该节点进行更新
        for(int j=1;j<=n;j++){
            if(dist[j]>dist[t]+g[t][j])
            {
                dist[j]=dist[t]+g[t][j];
            }
        }
     }
     if(dist[n]==0x3f3f3f3f)    return -1;
     return dist[n];
 }
 
 int main(){
     cin>>n>>m;
     memset(g,0x3f,sizeof g);
     while(m--){
         int x,y,z;
         cin>>x>>y>>z;
         if(g[x][y]>z)g[x][y]=z;
     }
     cout<<Dijkstra()<<endl;
     return 0;
 }

从核心步骤的迭代我们可以看出，朴素版dijkstra算法的时间复杂度是O(n^2)。

堆优化版dijkstra算法

即运用堆排序来加快找边的速度，使算法总体时间复杂度达到O（（n+m）logm）。总的来说，就是把图中所有边都压入一个最小堆（即根节点的值最小的堆）中，每次可以直接取堆顶元素入最短路，取边的时间复杂度为O（1），但维护最小堆的时间复杂度为O（log m)。

代码如下：

#include<bits/stdc++.h>
using namespace std;

typedef pair<int,int> PII;

const int N = 1e5+5e4+10;

int h[N],w[N],e[N],ne[N],idx;
int dis[N];
bool st[N];
int n,m;

void add(int a,int b,int c){
    e[idx]=b,w[idx]=c,ne[idx]=h[a];h[a]=idx++;
}

int dijkstra(){
    memset(dis,0x3f,sizeof(dis));
    dis[1]=0;
    priority_queue<PII,vector<PII>,greater<PII> > heap;//优先队列
    heap.push({0,1});
    
    while(heap.size())
    {
        auto t=heap.top();heap.pop();
        
        int distance=t.first,ver=t.second;
        if(st[ver]) continue;
            
        for(int i=h[ver];i!=-1;i=ne[i])
        {
            int j=e[i];
            if(dis[j]>distance+w[i])
            {
                dis[j]=distance+w[i];
                heap.push({dis[j],j});
            }
            
        }
        st[ver]=true;
    }
    return dis[n]==0x3f3f3f3f?-1:dis[n];
}

int main(void){
    cin>>n>>m;
    
    memset(h,-1,sizeof(h));
    while(m--)
    {
        int a,b,c;
        cin>>a>>b>>c;
        add(a,b,c);
    }
    cout<<dijkstra();
    return 0;
}

bellman—ford算法

Bellman - ford 算法是求含负权图的单源最短路径的一种算法，效率较低，代码难度较小。其原理为连续进行松弛，在每次松弛时把每条边都更新一下，若在 n-1 次松弛后还能更新，则说明图中有负环，因此无法得出结果，否则就完成。

时间复杂度为O（nm）。代码如下：

#include<bits/stdc++.h>
using namespace std;

const int N=5e2+10,M=1e4+10;
int n,m,k;

struct Edge{
    int a;
    int b;
    int c;
}e[M];
int dis[N];
int back[N];

int bellman_ford(){
    dis[1]=0;
    for(int i=0;i<k;i++)
    {
        memcpy(back,dis,sizeof(dis));
        for(int j=0;j<m;j++)
        {
            int a=e[j].a,b=e[j].b,c=e[j].c;
            dis[b]=min(dis[b],back[a]+c);
        }
    }
    return dis[n]>0x3f3f3f3f/2 ? -100: dis[n];
}

int main(void)
{
    cin>>n>>m>>k;
    memset(dis,0x3f,sizeof(dis));
    
    for(int i=0;i<m;i++)
    {
        int a,b,c;
        cin>>a>>b>>c;
        e[i]={a,b,c};
    }
    
    int res=bellman_ford();
    
    if(res!=-100)cout<<res;
    else cout<<"impossible";
    
    return 0;
}

SPFA算法

其实关于spfa算法，还有一段婉转镌刻的故事（bushi，详情各位自行百度。

spfa算法其实是Bellman—ford的队列优化形式，常用于求含负权边的单源最短路径，以及判断负环。优化过的情况下，一般的时间复杂度为O(m),但是在最坏情况下，其实是与朴素版Bellman—ford算法一样的复杂度。

代码如下：

#include<bits/stdc++.h>
using namespace std;

const int N =1e5+10;

int dis[N],h[N],e[N],ne[N],w[N],idx;
int n,m;
bool st[N];

void add(int a,int b,int c){
    e[idx]=b,ne[idx]=h[a],w[idx]=c,h[a]=idx++;
}

int spfa(){
    memset(dis,0x3f,sizeof(dis));
    queue<int>q;
    dis[1]=0;
    q.push(1);
    st[1]=true;
    
    while(q.size())
    {
        int t=q.front();
        q.pop();
        
        st[t]=false;
        for(int i=h[t];i!=-1;i=ne[i])
        {
            int j=e[i];
            if(dis[j]>dis[t]+w[i])
            {
                dis[j]=dis[t]+w[i];
                if(!st[j])
                {
                    q.push(j);
                    st[j]=true;
                }
            }
        }
    }
    
    return dis[n];
}

int main(void){
    cin>>n>>m;
    
    memset(h,-1,sizeof(h));
    while(m--)
    {
        int a,b,c;
        cin>>a>>b>>c;
        add(a,b,c);
    }
    int res=spfa();

    if(res!=0x3f3f3f3f)cout<<res;
    else cout<<"impossible";
    
    return 0;
}

值得一提的是，在有边数限制的情况下，spfa算法是最好的选择。
